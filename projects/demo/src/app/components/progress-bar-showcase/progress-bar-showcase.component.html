MerProgressBar Component

Basic Usage

HTML


<mer-progress-bar
  [value]="progressValue"
  [indeterminate]="isIndeterminate">
</mer-progress-bar>

TypeScript


progressValue = 0.5;
isIndeterminate = false;

Practical Example: Download Simulation

example-file.pdf

Status: {{ isDownloading ? 'Downloading...' : (downloadProgress > 0 ? 'Downloaded' : 'Ready') }}

HTML


<div class="download-demo">
  <div class="file-info">
    <h4>example-file.pdf</h4>
    <p>Status: {{ isDownloading ? 'Downloading...' : (downloadProgress > 0 ? 'Downloaded' : 'Ready') }}</p>
  </div>

  <mer-progress-bar
    [value]="downloadProgress"
    [indeterminate]="false">
  </mer-progress-bar>

  <button
    (click)="startDownload()"
    [disabled]="isDownloading">
            {{ isDownloading ? 'Downloading...' : 'Start Download' }}
            </button>
</div>

TypeScript


downloadProgress = 0;
isDownloading = false;
